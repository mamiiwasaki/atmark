Array

                unshift         // 先頭に追加
                push            // 末尾に追加
                shift           // 先頭から削除
                pop             // 末尾から削除
                splice          // 途中に追加

                var a = new Array(100, 300, 200);
                console.log(a.length);
                a.push(500);		// 末尾に追加
                console.log(a);		// [100, 300, 200, 500]
                a.splice(1, 2);		// 1番目から2つ削除
                console.log(a);		// [100, 500]
                a.splice(1, 2, 800, 1000);		// 1番目から2つ削除し、その場所から追加
                console.log(a);		// [100, 800, 1000]

おみくじの確率を操作

				var btn = document.getElementById('btn');
				btn.addEventListener('click', function(){
                    // ----------------------------------------
                    // ランダムに出す
                    var results = ['very good','good','bad','very bad'];
					var n = Math.floor(Math.random()*results.length);
					this.textContent = results[n];

                    // ----------------------------------------
                    // 確率を操作する
					var n = Math.random();
					if(n < 0.05){
						this.textContent = 'very good';		// 5%の確率
					} else if(n < 0.2) {
						this.textContent = ' good';			// 15%の確率
					} else {
						this.textContent = 'bad';			// 80%の確率
					}
				});
