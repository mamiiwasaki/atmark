                       ========================
        セレクタ
        html要素
        id		#
        class	.
        > 	直下の要素
        (space) それ以下の要素
        .	複数の要素
        +	隣接する要素
        $('p, .item').css('color', 'red');
        $('.item + .item').css('color', 'red');

        ========================
        フィルタ
        :eq()
        :gt(), :li()		greater, less
        :even, :odd
        :conteints()
        :first, :last
        $('#sub > li:gt(2)').siblings().css('color', 'red');

        ========================
        メソッドを使った DOM要素の指定
        parent(), children()
                next(), prev()
        siblings()	- 兄弟要素
        $('#sub > li:eq(2)').siblings().css('color', 'red');

        ========================
        属性セレクタ
        [] で囲む
        =			イコール
        !=			以外
        *=			を含む
        ^=			から始まる
        $=			で終わる
        $('a[href$=jp]').css('color', 'red');

        メソッド
        .css	設定、取得
            $('p').css('color', 'red');

        メッソッドチェーン
        $('p').css('color', 'red').css('background', 'blue');

        attrメソッド 属性　取得、設定
            var url = $('a').attr('href');
            $('a').attr('href', 'http://www.google.co.jp');

        dataメソッド
            var dat = $('a').data('sitename');

        -- エフェクトのメソッド
        hide, show, fadeOut, fadeIn, toggle
        $('#box').hide();
        $('#box').hide('slow');	// ゆっくり隠す
        $('#box').hide(800);	// 0.8秒かけて
        $('#box').fadeOut(800);
        $('#box').toggle(800);		// 表示されていたら隠し、隠れていたら表示
        ★★★callback関数も指定できる！！処理が終わったら実行される
        $('#box').fadeOut(2000, function(){
            alert('hello');
        });

        ========================
        イベント
        click, mouseover, mouseover, mousemove

        メソッドチェーンも使える!!
        $('#box')
                .mouseover(function(){
                       $(this).css('background', 'green');
                }).mouseout(function(){
                    $(this).css('background', '');
                }).mousemove(function(e){
                    $(this).text(e.pageX);
            });

        //eventオブジェクト

            keypress	keyup	keydown	dragdrop	move	resize

                }).blur(function(){
                    $(this).css('background', '');
