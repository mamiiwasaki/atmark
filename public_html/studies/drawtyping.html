								window.addEventListener('load', function(){
									drawTyping('.typingDisplay');
								});
								function drawTyping(attribute){
									Array.prototype.forEach.call(document.querySelectorAll(attribute), function(node){
										var i=0;
										var baseStr = node.innerHTML;
										var newStr = '';
										var timer_id = setInterval(function(){
											if(i>baseStr.length){
												clearInterval(timer_id);
											}
											newStr += baseStr.charAt(i);
											node.innerHTML = newStr;
											i++;
										}, 50);
									});
								}
