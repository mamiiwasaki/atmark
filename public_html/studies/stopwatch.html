var h=0;
var m=0;
var s=0;
var ms=0;
var timer_id ;
var clock = document.getElementById('clock');
var btn_start = document.getElementById('btn_start');
var btn_stop = document.getElementById('btn_stop');
var btn_restart = document.getElementById('btn_restart');
var btn_reset = document.getElementById('btn_reset');

// start
btn_start.addEventListener('click', function(){
	if(this.classList.contains('btn_disabled')){
		return;
	}
	timer_id = setInterval(countup, 10);
	this.classList.add('btn_disabled');
	btn_stop.classList.remove('btn_disabled');
	btn_reset.classList.remove('btn_disabled');
});
// stop
btn_stop.addEventListener('click', function(){
	if(this.classList.contains('btn_disabled')){
		return;
	}
	clearInterval(timer_id);
	this.classList.add('btn_disabled');
	btn_restart.classList.remove('btn_disabled');
});
// restart
btn_restart.addEventListener('click', function(){
	if(this.classList.contains('btn_disabled')){
		return;
	}
	timer_id = setInterval(countup, 10);
	this.classList.add('btn_disabled');
	btn_stop.classList.remove('btn_disabled');
});
// reset
btn_reset.addEventListener('click', function(){
	if(this.classList.contains('btn_disabled')){
		return;
	}
	h=0;
	m=0;
	s=0;
	sf=0;
	clock.innerHTML = zeroume(h)+':'+zeroume(m)+':'+zeroume(s)+'.'+zeroume(sf);
	clearInterval(timer_id);
	this.classList.add('btn_disabled');
	btn_stop.classList.add('btn_disabled');
	btn_restart.classList.add('btn_disabled');
	btn_start.classList.remove('btn_disabled');
});
// countup
function countup(){
	ms++;
	if(ms>99){
		s++;
		ms=0;
	}
	if(s>59){
		m++;
		s=0;
	}
	if(m>59){
		h++;
		m=0;
	}
	clock.innerHTML = zeroume(h)+':'+zeroume(m)+':'+zeroume(s)+'.'+zeroume(ms);
}
// ゼロ埋め
function zeroume(t){
	return ('0'+t).slice(-2);
}
