								var data = [1,2,3,4,5,6,7,8,9];
								var tbl = document.getElementById('tbl');
								//var x, y, i, j;

								// header
								var tr = document.createElement('tr');
								tr.innerHTML = '<th></th>';
								for(var y=0; y<9; y++){
									tr.innerHTML += '<th>'+data[y]+'</th>';
								}
								tbl.appendChild(tr);

								// kuku
								for(var x=0; x<data.length; x++){
									// tr
									var tr = document.createElement('tr');
									tr.innerHTML = '<th>'+data[x]+'</th>';

									for(var y=0; y<data.length; y++){
										var td = document.createElement('td');
										td.innerHTML = data[x]*data[y];
										tr.appendChild(td);
									}
									tbl.appendChild(tr);
								}
